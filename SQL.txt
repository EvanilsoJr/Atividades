create table aluno (
	id integer not null auto_increment primary key,
    nome varchar(60) not null,
    data_nascimento date,
    cpf varchar(11)
);

create table professor (
	id integer not null auto_increment primary key,
    nome varchar(60) not null,
    salario decimal(10,2)
);

create table coordenador (
	id integer not null auto_increment primary key,
    nome varchar(60) not null,
    salario decimal(10,2)
);

create table curso (
	id integer not null auto_increment primary key,
    nome varchar(60) not null,
    valor decimal(10,2) not null,
    carga_horaria smallint,
    id_coordenador integer unique not null,
    foreign key(id_coordenador) references coordenador(id)
);

create table turma (
	id integer not null auto_increment primary key,
    nome varchar(60) not null,
    id_curso integer not null,
    foreign key(id_curso) references curso(id)
);

create table matricula (
	id integer not null auto_increment primary key,
    numero varchar(10) not null,
    id_aluno integer unique not null,
    id_turma integer not null,
    foreign key(id_aluno) references aluno(id),
    foreign key(id_turma) references turma(id)
);

create table professor_turma (
	id_professor integer not null,
    id_turma integer not null,
    primary key(id_professor, id_turma),
    foreign key(id_professor) references professor(id),
    foreign key(id_turma) references turma(id)
);

create table disciplina (
	id integer not null auto_increment primary key,
    nome varchar(60) not null,
    carga_horaria smallint,
    id_professor integer not null,
    id_curso integer not null,
    foreign key(id_professor) references professor(id),
    foreign key(id_curso) references curso(id)
);

create table avaliacao (
	id integer not null auto_increment primary key,
    nome varchar(20) not null,
    pontuacao_total decimal(3,1) not null,
    id_disciplina integer not null,
    foreign key(id_disciplina) references disciplina(id)
);

create table complementar (
	id integer not null auto_increment primary key,
    certificado blob not null,
    id_aluno integer not null,
    foreign key(id_aluno) references aluno(id)
);
